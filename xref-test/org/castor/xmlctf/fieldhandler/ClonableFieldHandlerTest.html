19    @Test
20    public void unmarshalWithClonableFieldHandler()
21        throws MarshalException, ValidationException, IOException, MappingException {
22      Mapping mapping = new Mapping();
23      mapping.loadMapping(new InputSource(this.getClass().getResource("mapping.xml").openStream()));
24  
25      XMLContext context = new XMLContext();
26      context.addMapping(mapping);
27  
28      Unmarshaller unmarshaller = context.createUnmarshaller();
29      unmarshaller.setClass(Root.class);
30      Root root = (Root) unmarshaller
31          .unmarshal(new StringReader("<?xml version=\"1.0\"?><root><date>20040510</date></root>"));
32  
33      assertNotNull(root);
34      assertNotNull(root.getDate());
35  
36    }
