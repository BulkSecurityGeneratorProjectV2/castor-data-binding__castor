59      public static Test suite() {
60          TestSuite suite = new TestSuite("Base64Encoder Tests");
61  
62          suite.addTest(new TestBase64Encoder("testAllBytes"));
63          suite.addTest(new TestBase64Encoder("testPadding"));
64  
65          return suite;
66      }
67  
68      public TestBase64Encoder(final String name) { super(name); }
69  
70      public void testAllBytes() {
71          for (int i = 0; i < 256; i++) {
72              assertEquals(MAP[i], new String(Base64Encoder.encode(new byte[] {(byte) i})));
73          }
74      }
75  
76      public void testPadding() {
77          String str;
78  
79          str = new String(Base64Encoder.encode(new byte[] {33}));
80          assertEquals("IQ==", str);
81  
82          str = new String(Base64Encoder.encode(new byte[] {33, 33}));
83          assertEquals("ISE=", str);
84  
85          str = new String(Base64Encoder.encode(new byte[] {33, 33, 33}));
86          assertEquals("ISEh", str);
87  
88          str = new String(Base64Encoder.encode(new byte[] {33, 33, 33, 34}));
89          assertEquals("ISEhIg==", str);
90      }
