48    @Test
49    public void testAllBytes() {
50      for (int i = 0; i < 256; i++) {
51        byte[] content = Base64Decoder.decode(MAP[i]);
52        assertTrue(1 == content.length);
53        if (i < 128) {
54          assertTrue(i == content[0]);
55        } else {
56          assertTrue((i - 256) == content[0]);
57        }
58      }
59    }
60  
61    @Test
62    public void testPadding() {
63      byte[] content;
64      content = Base64Decoder.decode("IQ==");
65      assertTrue(1 == content.length);
66      assertTrue(33 == content[0]);
67  
68      content = Base64Decoder.decode("ISE=");
69      assertTrue(2 == content.length);
70      assertTrue(33 == content[0]);
71      assertTrue(33 == content[1]);
72  
73      content = Base64Decoder.decode("ISEh");
74      assertTrue(3 == content.length);
75      assertTrue(33 == content[0]);
76      assertTrue(33 == content[1]);
77      assertTrue(33 == content[2]);
78  
79      content = Base64Decoder.decode("ISEhIg==");
80      assertTrue(4 == content.length);
81      assertTrue(33 == content[0]);
82      assertTrue(33 == content[1]);
83      assertTrue(33 == content[2]);
84      assertTrue(34 == content[3]);
85    }
