16  package org.castor.xml.schema.namespace;
17  
18  import static org.junit.Assert.assertEquals;
19  
20  import org.castor.xml.schema.ComparisResultExtractor;
21  import org.castor.xml.schema.ComparisonResult;
22  import org.exolab.castor.xml.schema.Schema;
23  import org.junit.Ignore;
24  import org.junit.Test;
25  
26  /**
27   * Namespace tests.
28   * 
29   * @author <a href="mailto:leducbao AT gmail DOT com">Le Duc Bao</a>
30   */
31  public final class NamespaceTest {
32  
33      /**
34       * Create a schema with single namespace
35       * 
36       * @throws Exception
37       */
38      @Test
39      public void testSingleNamespace() throws Exception {
40  
41          // create a new XML schema representation.
42          Schema schema = new Schema();
43          
44          // create targeted schema
45          schema.addNamespace("myprefix", "my.namespace.org");
46  
47          ComparisonResult result = ComparisResultExtractor.doTest(schema, this.getClass().getResource("namespace_singlenamespace.xsd"));
48          assertEquals("single namespace add failed", ComparisonResult.IDENTICAL,
49                  result);
50      }
51  
52      /**
53       * Create a schema with single namespace
54       * @throws Exception 
55       */
56      @Ignore
57      public void testDiff() throws Exception {
58  
59          // create a new XML schema representation.
60          Schema schema = new Schema();
61          
62          // create targeted schema
63          schema.addNamespace("wrong", "my.namespace.org");
64  
65          ComparisonResult result = ComparisResultExtractor.doTest(schema, this.getClass().getResource("namespace_singlenamespace.xsd"));
66          assertEquals("test diff", ComparisonResult.DIFFERENCE, result);
67      }
68  
69      /**
70       * Test for multiple namespaces
71       * @throws Exception 
72       */
73      @Test
74      public void testMultipleNamespace() throws Exception {
75  
76          // create a new XML schema representation.
77          Schema schema = new Schema();
78          
79          // create targeted schema
80          schema.addNamespace("myprefix", "my.namespace.org");
81          schema.addNamespace("other", "other.namespace.org");
82  
83          ComparisonResult result = ComparisResultExtractor.doTest(schema, this.getClass().getResource("namespace_multiplenamespace.xsd"));
84          assertEquals("multiple namespace add failed", ComparisonResult.IDENTICAL,
85                  result);
86      }
87  }
