18  import org.exolab.castor.xml.schema.AttributeDecl;
19  import org.exolab.castor.xml.schema.Facet;

22  /**
23   * This test covers simple type generation.
24   * 
25   * @author <a href="mailto:leducbao AT gmail DOT com">Le Duc Bao</a>
26   */
27  public class SimpleTypeTest extends AbstractSchemaTest {
28  
29      /**
30       * @param Constructor
31       */
32      public SimpleTypeTest(String testcase) {
33          super(testcase);
34      }
35  
36      /**
37       * very simple type
38       */
39      public void testSimpleType() throws Exception  {
40  
41          // create targeted schema
42          _schema.addNamespace("pre", "my.namespace.org");
43          SimpleType sType = _schema.createSimpleType("myType", "string", "");
44  
45          _schema.addSimpleType(sType);
46  
47          // compare
48          TestResult result = doTest("simpletype_simple.xsd");
49          assertEquals("single attribute test failed", TestResult.IDENTICAL,
50                  result);
51      }
52  
53      /**
54       * test create attribute, fixed value
55       */
56      public void testAttributeCreation() throws Exception  {
57  
58          // create targeted schema
59          _schema.addNamespace("pre", "my.namespace.org");
60  
61          AttributeDecl attr = new AttributeDecl(_schema);
62          attr.setName("myAttr");
63          attr.setSimpleTypeReference("string");
64          attr.setFixedValue("#hello");
65          attr.setUse(AttributeDecl.USE_OPTIONAL);
66          _schema.addAttribute(attr);
67  
68          // compare
69          TestResult result = doTest("simpletype_attributecreation.xsd");
70          assertEquals("testAttributeCreation test failed", TestResult.IDENTICAL,
71                  result);
72      }
73

118     // restriction
119     /**
120      * test create facet/min-max
121      */
122     public void testMinMax() throws Exception  {
123 
124         // create targeted schema
125         _schema.addNamespace("pre", "my.namespace.org");
126 
127         SimpleType sType = _schema.createSimpleType("myType", "int", "");
128 
129         Facet min = new Facet(Facet.MIN_EXCLUSIVE, "0");
130         Facet max = new Facet(Facet.MAX_EXCLUSIVE, "100");
131         sType.addFacet(min);
132         sType.addFacet(max);
133 
134         _schema.addSimpleType(sType);
135 
136         // compare
137         TestResult result = doTest("simpletype_res_minmax.xsd");
138         assertEquals("testMinMax test failed", TestResult.IDENTICAL, result);
139     }

143     // enumeration
144     // union
145     // pattern
146     // precision, total digits, fraction digits
