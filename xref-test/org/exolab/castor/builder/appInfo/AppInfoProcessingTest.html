69    /**
70     * This method tests the processing of AppInfos from a very basic Schema file. The Schema file
71     * contains a single <code>element</code> definition and doesen't contain a top-level, named
72     * complex type. The element doesen't have any attributes either.
73     * 
74     * @throws Exception if an error occurs.
75     */
76    public final void testAppInfoProcessingWithSimpleSchema() throws Exception {
77      _xmlSchema = getClass().getResource("simple.xsd").toExternalForm();
78      InputSource inputSource = new InputSource(_xmlSchema);
79      _generator.generateSource(inputSource, getClass().getPackage().getName() + ".generated.simple");
80  
81      // get the Source Generator's state Info
82      SGStateInfo sgState = _generator.getSGStateInfo();
83  
84      assertNotNull(sgState);
85  
86      /*
87       * Get all elements (=keys) the ClassInfoResolver knows of. For these keys the SourceGenerator
88       * created ClassInfos in which we are interested.
89       */
90      Enumeration enumeration = sgState.keys();
91  
92      assertTrue(enumeration.hasMoreElements());
93  
94      List cInfos = new ArrayList();
95      /*
96       * Get all ClassInfos. Note that during the source generation process ClassInfos are actually
97       * added twice to the ClassInfoResolver's cache: - once with the XMLBindingComponent/ClassInfo -
98       * and with the JClass/ClassInfo as key/value pair. Therefore we only get those ClassInfos with
99       * a XMLBindingComponent/ClassInfo pair to avoid duplicates!
100      */
101     while (enumeration.hasMoreElements()) {
102       Object elem = enumeration.nextElement();
103       if (!(elem instanceof JClass)) {
104         cInfos.add(sgState.resolve(elem));
105       }
106     }
107 
108     assertEquals(1, cInfos.size());
109 
110     ClassInfo cInfo = (ClassInfo) cInfos.get(0);
111     XMLInfoNature xmlNature = new XMLInfoNature(cInfo);
112 
113     assertEquals("father", xmlNature.getNodeName());
114 
115     JDOClassInfoNature cNature = new JDOClassInfoNature(cInfo);
116 
117     assertEquals("Father", cNature.getTableName());
118 
119     List primaryKeys = cNature.getPrimaryKeys();
120 
121     assertEquals(1, primaryKeys.size());
122     assertEquals("ssnr", primaryKeys.get(0));
123 
124     ///////// FieldInfo
125     /*
126      * There should be 3 FieldInfos: - ssnr - firstName - lastName
127      */
128     assertEquals(3, cInfo.getFieldCount());
129 
130     FieldInfo fInfo = cInfo.getElementField("ssnr");
131 
132     assertNotNull(fInfo);
133 
134     JDOFieldInfoNature fNature = new JDOFieldInfoNature(fInfo);
135 
136     assertEquals("ssnr", fNature.getColumnName());
137     assertEquals("integer", fNature.getColumnType());
138 
139     fInfo = cInfo.getElementField("firstName");
140 
141     assertNotNull(fInfo);
142 
143     fNature = new JDOFieldInfoNature(fInfo);
144 
145     assertEquals("firstName", fNature.getColumnName());
146     assertEquals("varchar", fNature.getColumnType());
147 
148     fInfo = cInfo.getElementField("lastName");
149 
150     assertNotNull(fInfo);
151 
152     fNature = new JDOFieldInfoNature(fInfo);
153 
154     assertEquals("lastName", fNature.getColumnName());
155     assertEquals("varchar", fNature.getColumnType());
156 
157     try {
158       assertNull(cInfos.get(1));
159       fail("Unexpected ClassInfo Element encountered!");
160     } catch (IndexOutOfBoundsException e) {
161 
162     }
163   }

165   /**
166    * This method tests the processing of AppInfos from a Schema that describes a Book. The Schema
167    * consists of a global element whose type is described by a global <code>complexType</code>
168    * definition consisting of two sub-elements and an attribute.
169    * 
170    * @throws Exception if an error occurs
171    */
172   public final void testAppInfoProcessingWithEntitySchema() throws Exception {
173     _xmlSchema = getClass().getResource("schema-entity.xsd").toExternalForm();
174     InputSource inputSource = new InputSource(_xmlSchema);
175     _generator.generateSource(inputSource, getClass().getPackage().getName() + ".generated.entity");
176 
177     // get the Source Generator's state Info
178     SGStateInfo sgState = _generator.getSGStateInfo();
179 
180     assertNotNull(sgState);
181 
182     /*
183      * Get all elements (=keys) the ClassInfoResolver knows of. For these keys the SourceGenerator
184      * created ClassInfos in which we are interested.
185      */
186     Enumeration enumeration = sgState.keys();
187 
188     assertTrue(enumeration.hasMoreElements());
189 
190     List cInfos = new ArrayList();
191     /*
192      * Get all ClassInfos. Note that during the source generation process ClassInfos are actually
193      * added twice to the ClassInfoResolver's cache: - once with the XMLBindingComponent/ClassInfo -
194      * and with the JClass/ClassInfo as key/value pair. Therefore we only get those ClassInfos with
195      * a XMLBindingComponent/ClassInfo pair to avoid duplicates!
196      */
197     while (enumeration.hasMoreElements()) {
198       Object elem = enumeration.nextElement();
199       if (!(elem instanceof JClass)) {
200         cInfos.add(sgState.resolve(elem));
201       }
202     }
203 
204     assertEquals(2, cInfos.size());
205 
206     for (int i = 0; i < cInfos.size(); ++i) {
207       ClassInfo cInfo = (ClassInfo) cInfos.get(i);
208 
209       assertNotNull(cInfo);
210 
211       XMLInfoNature xmlNature = new XMLInfoNature(cInfo);
212 
213       if (xmlNature.getNodeName().equals("book")) {
214         /*
215          * No JDO-specific information should be stored to this ClassInfo.
216          */
217         assertFalse(cInfo.hasNature(JDOClassInfoNature.class.getName()));
218         // JDOClassNature cNature = new JDOClassNature(cInfo);
219         //
220         // assertEquals(null, cNature.getTableName());
221         //
222         // List primaryKeys = cNature.getPrimaryKeys();
223         //
224         // assertNull(primaryKeys);
225 
226         assertEquals(0, cInfo.getFieldCount());
227 
228       } else if (xmlNature.getNodeName().equals("bookType")) {
229         JDOClassInfoNature cNature = new JDOClassInfoNature(cInfo);
230 
231         assertEquals("book", cNature.getTableName());
232 
233         List primaryKeys = cNature.getPrimaryKeys();
234 
235         assertEquals(1, primaryKeys.size());
236         assertEquals("isbn", (String) primaryKeys.get(0));
237 
238         ///////// FieldInfo
239         /*
240          * There should be 3 FieldInfos: - coverType (attribute field) - isbn - title
241          */
242         assertEquals(3, cInfo.getFieldCount());
243 
244         // FieldInfo for coverType
245         FieldInfo fInfo = cInfo.getAttributeField("coverType");
246 
247         assertNotNull(fInfo);
248 
249         fInfo.addNature(JDOFieldInfoNature.class.getName());
250         JDOFieldInfoNature fNature = new JDOFieldInfoNature(fInfo);
251 
252         assertEquals("cover_type", fNature.getColumnName());
253         assertEquals("varchar", fNature.getColumnType());
254 
255         // FieldInfo for isbn
256         fInfo = cInfo.getElementField("isbn");
257 
258         assertNotNull(fInfo);
259 
260         fNature = new JDOFieldInfoNature(fInfo);
261 
262         assertEquals("isbn", fNature.getColumnName());
263         assertEquals("varchar", fNature.getColumnType());
264 
265         // FieldInfo for title
266         fInfo = cInfo.getElementField("title");
267 
268         assertNotNull(fInfo);
269 
270         fNature = new JDOFieldInfoNature(fInfo);
271 
272         assertEquals("title", fNature.getColumnName());
273         assertEquals("varchar", fNature.getColumnType());
274       } else {
275         fail("Unexpected ClassInfo Element encountered!");
276       }
277     }

469   public final void testAppInfoProcessingWithComplexContent() throws Exception {
470     _xmlSchema = getClass().getResource("complex-content.xsd").toExternalForm();
471     InputSource inputSource = new InputSource(_xmlSchema);
472     _generator.generateSource(inputSource,
473         getClass().getPackage().getName() + ".generated.complexContent");
474 
475     // get the Source Generator's state Info
476     SGStateInfo sgState = _generator.getSGStateInfo();
477 
478     assertNotNull(sgState);
479 
480     /*
481      * Get all elements (=keys) the ClassInfoResolver knows of. For these keys the SourceGenerator
482      * created ClassInfos in which we are interested.
483      */
484     Enumeration enumeration = sgState.keys();
485 
486     assertTrue(enumeration.hasMoreElements());
487 
488     List cInfos = new ArrayList();
489     /*
490      * Get all ClassInfos. Note that during the source generation process ClassInfos are actually
491      * added twice to the ClassInfoResolver's cache: - once with the XMLBindingComponent/ClassInfo -
492      * and with the JClass/ClassInfo as key/value pair. Therefore we only get those ClassInfos with
493      * a XMLBindingComponent/ClassInfo pair to avoid duplicates!
494      */
495     while (enumeration.hasMoreElements()) {
496       Object elem = enumeration.nextElement();
497       if (!(elem instanceof JClass)) {
498         cInfos.add(sgState.resolve(elem));
499       }
500     }
501 
502     assertEquals(2, cInfos.size());
503 
504     for (int i = 0; i < cInfos.size(); ++i) {
505       ClassInfo cInfo = (ClassInfo) cInfos.get(i);
506 
507       assertNotNull(cInfo);
508 
509       JDOClassInfoNature cNature = new JDOClassInfoNature(cInfo);
510       XMLInfoNature xmlNature = new XMLInfoNature(cInfo);
511 
512       assertNotNull(cNature);
513 
514       if (xmlNature.getNodeName().equals("person")) {
515         String tableName = cNature.getTableName();
516         assertEquals("person", tableName);
517 
518         List primaryKeys = cNature.getPrimaryKeys();
519 
520         assertEquals(1, primaryKeys.size());
521         assertEquals("ssn", (String) primaryKeys.get(0));
522 
523         assertEquals(4, cInfo.getFieldCount());
524 
525         FieldInfo[] fInfos = cInfo.getElementFields();
526 
527         assertNotNull(fInfos);
528 
529         for (int j = 0; j < fInfos.length; ++j) {
530           FieldInfo fInfo = fInfos[j];
531           assertNotNull(fInfo);
532           JDOFieldInfoNature fNature = new JDOFieldInfoNature(fInfo);
533           assertNotNull(fNature);
534 
535           String columnName = fNature.getColumnName();
536           String columnType = fNature.getColumnType();
537 
538           XMLInfoNature xmlFieldNature = new XMLInfoNature(fInfo);
539 
540           if (xmlFieldNature.getNodeName().equals("ssn")) {
541             assertEquals("ssn", columnName);
542             assertEquals("bigint", columnType);
543           } else if (xmlFieldNature.getNodeName().equals("firstName")) {
544             assertEquals("firstName", columnName);
545             assertEquals("varchar", columnType);
546           } else if (xmlFieldNature.getNodeName().equals("lastName")) {
547             assertEquals("lastName", columnName);
548             assertEquals("varchar", columnType);
549           } else if (xmlFieldNature.getNodeName().equals("birthdate")) {
550             assertEquals("birthdate", columnName);
551             assertEquals("date", columnType);
552           } else {
553             fail("Unexpected FieldInfo Element encountered!");
554           }
555         }
556       } else if (xmlNature.getNodeName().equals("insurant")) {
557         String tableName = cNature.getTableName();
558         assertEquals("insurant", tableName);
559 
560         List primaryKeys = cNature.getPrimaryKeys();
561 
562         assertEquals(2, primaryKeys.size());
563         assertEquals("ssn", (String) primaryKeys.get(0));
564         assertEquals("policyNumber", (String) primaryKeys.get(1));
565 
566         assertEquals(1, cInfo.getFieldCount());
567 
568         FieldInfo fInfo = cInfo.getElementField("policyNumber");
569 
570         assertNotNull(fInfo);
571 
572         JDOFieldInfoNature fNature = new JDOFieldInfoNature(fInfo);
573 
574         assertNotNull(fNature);
575 
576         String columnName = fNature.getColumnName();
577         assertEquals("policyNumber", columnName);
578         String columnType = fNature.getColumnType();
579         assertEquals("bigint", columnType);
580       } else {
581         fail("Unexpected ClassInfo Element encountered!");
582       }
583     }
