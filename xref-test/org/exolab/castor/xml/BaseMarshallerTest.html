372   /**
373    * Asserts the values of xpath expression evaluation is exactly the same as expected value.
374    * </p>
375    * The xpath may contain the xml namespace prefixes, since namespaces from emails example are
376    * being registered.
377    *
378    * @param msg the error message that will be used in case of test failure
379    * @param expected the expected value
380    * @param xpath the xpath to evaluate
381    * @param xmlDoc the xml to use
382    * @throws Exception if any error occurs during xpath evaluation
383    */
384   private void assertXpathEvaluatesTo(String msg, String expected, String xpath, String xmlDoc)
385       throws Exception {
386     Map<String, String> namespaces = new HashMap<String, String>();
387     namespaces.put("tns", "http://castor.org/email");
388     namespaces.put("xsi", "http://www.w3.org/2001/XMLSchema-instance");
389 
390     NamespaceContext ctx = new SimpleNamespaceContext(namespaces);
391     XpathEngine engine = XMLUnit.newXpathEngine();
392     engine.setNamespaceContext(ctx);
393 
394     Document doc = XMLUnit.buildControlDocument(xmlDoc);
395     NodeList node = engine.getMatchingNodes(xpath, doc);
396 
397     assertEquals(msg, expected, node.item(0).getNodeValue());
398   }

433   /**
434    * Creates new instance of Marshaller.
435    *
436    * @param mapping the mapping file
437    * @return configured Marshaller instance
438    * @throws org.exolab.castor.mapping.MappingException if any error occurs when loading mapping
439    *         file
440    */
441   private Marshaller createMarshallerFromMapping(String mapping) throws MappingException {
442     XMLContext xmlContext = createXmlContextFromMapping(mapping);
443 
444     Marshaller marshaller = xmlContext.createMarshaller();
445     marshaller.setValidation(true);
446     return marshaller;
447   }

449   /**
450    * Creates a XMLContext with loaded mapping file.
451    *
452    * @param mappingPath the path to mapping file
453    * @return configured XMLContext
454    * @throws MappingException if any error occurs while loading mapping file.
455    */
456   private XMLContext createXmlContextFromMapping(String mappingPath) throws MappingException {
457     InputStream mappingFile = getClass().getResourceAsStream(mappingPath);
458 
459     XMLContext xmlContext = new XMLContext();
460     Mapping mapping = new Mapping();
461     mapping.loadMapping(new InputSource(mappingFile));
462     xmlContext.addMapping(mapping);
463     return xmlContext;
464   }
