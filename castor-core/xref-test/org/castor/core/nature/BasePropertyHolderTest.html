5   import junit.framework.TestCase;

7   /**
8    * Tests the set/get capabilities of a PropertyHolder. Subclasses MUST override
9    * the setUp method and set a PropertyHolder to _holder. The testcase needs a
10   * "fresh" NatureExtendable to guarantee independence for each test.
11   * 
12   * @author Lukas Lang
13   * @since 1.2.1
14   * 
15   */
16  public abstract class BasePropertyHolderTest extends TestCase {
17  
18      /**
19       * Test PropertyHolder to test. Must be set in subclass.
20       */
21      protected PropertyHolder _holder;
22  
23      /**
24       * Tests the holder's capability to hold properties.
25       */
26      public final void testSimpleReadWrite() {
27          _holder.setProperty("tablename", "BOOK");
28          assertEquals("BOOK", (String) _holder.getProperty("tablename"));
29      }
30  
31      /**
32       * Test verifies the postcondition of setTableName.
33       */
34      public final void testSetPropertyTwice() {
35          _holder.setProperty("tablename", "BOOK");
36          _holder.setProperty("tablename", "BOOKD");
37          assertEquals("BOOKD", (String) _holder.getProperty("tablename"));
38      }
39  
40      /**
41       * Tests default behavior of properties not set before.
42       */
43      public final void testGetButNotSetBefore() {
44          assertEquals(null, _holder.getProperty("tablename"));
45      }
46  }
