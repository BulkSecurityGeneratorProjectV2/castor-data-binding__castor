25  /**
26   * The base XML Schema Type class.
27   * 
28   * @author <a href="mailto:keith AT kvisco DOT com">Keith Visco</a>

32  public abstract class XSType {

34      
35      // special types

38      public static final short NULL = -1;

40      /** A user-defined type. */

61      public static final short GYEARMONTH_TYPE = 6;
62      
63      /** xsd:gYear. */
64      public static final short GYEAR_TYPE = 7;
65      
66      /** xsd:gMonthDay. */
67      public static final short GMONTHDAY_TYPE = 8;
68      
69      /** xsd:gDay. */
70      public static final short GDAY_TYPE = 9;
71      
72      /** xsd:gMonth. */
73      public static final short GMONTH_TYPE = 10;

75      /** xsd:boolean. */
76      public static final short BOOLEAN_TYPE = 11;
77      
78      /** xsd:base64Binary. */
79      public static final short BASE64BINARY_TYPE = 12;
80      
81      /** xsd:hexBinary. */
82      public static final short HEXBINARY_TYPE = 13;
83      
84      /** xsd:float. */
85      public static final short FLOAT_TYPE = 14;
86      
87      /** xsd:double. */
88      public static final short DOUBLE_TYPE = 15;

94      public static final short ANYURI_TYPE = 17;

97      public static final short QNAME_TYPE = 18;

99      /** xsd:notation. */
100     public static final short NOTATION_TYPE = 19;

103     
104     /** xsd:normalizedString. */
105     public static final short NORMALIZEDSTRING_TYPE = 20;

111     public static final short LANGUAGE_TYPE = 22;

182     //--------------------------------------------------------------------------
183 
184     /** Flag signaling an enumerated type. */
185     private boolean _enumerated = false;
186 
187     //--------------------------------------------------------------------------
188 
189     /**
190      * Returns true if this XSType represents an enumerated type.
191      * 
192      * @return True if this XSType represents an enumerated type.
193      */
194     public final boolean isEnumerated() {
195         return _enumerated;
196     }
197 
198     /**
199      * Sets the enumerated flag for this XSClass.
200      *
201      * @param enumerated A boolean indicating whether or not this XSClass represents an
202      *        enumerated type.
203      */
204     public final void setAsEnumerated(final boolean enumerated) {
205         _enumerated = enumerated;
206     }
207 
208     //--------------------------------------------------------------------------
209 
210     /**
211      * Returns the name of this XSType.
212      * 
213      * @return The name of this XSType.
214      */
215     public abstract String getName();
216 
217     /**
218      * Returns the type of this XSType.
219      * 
220      * @return the type of this XSType.
221      */
222     public abstract short getType();
223 
224     /**
225      * Returns true if this XSType represents a primitive type.
226      * 
227      * @return True if this XSType represents a primitive type.
228      */
229     public abstract boolean isPrimitive();
230     
231     /**
232      * Returns true if the XSType represents an XML Schema date/time type.
233      * 
234      * @return True if the XSType represents an XML Schema date/time type.
235      */
236     public abstract boolean isDateTime();
237 
238     /**
239      * Returns true if this XSType represents a collection.
240      * 
241      * @return True if this XSType represents a collection.
242      */
243     public boolean isCollection() {
244         return false;
245     }
246     
247     /**
248      * Returns the JType that this XSType represents.
249      * 
250      * @return The JType that this XSType represents.
251      */
252     public abstract JType getJType();
253 
254     /**
255      * Returns the Java code neccessary to create a new instance of the JType
256      * associated with this XSType.
257      *
258      * @return The Java code neccessary to create a new instance.
259      */
260     public abstract String newInstanceCode();
261 
262     /**
263      * Returns the string necessary to convert an instance of this XSType to an
264      * Object. This method is really only useful for primitive types.
265      *
266      * @param variableName The name of the instance variable.
267      * @return The String necessary to convert an instance of this XSType to an Object.
268      */
269     public abstract String createToJavaObjectCode(final String variableName);
270 
271     /**
272      * Returns the string necessary to convert an Object to an instance of this
273      * XSType. This method is really only useful for primitive types.
274      *
275      * @param variableName The name of the Object.
276      * @return The String necessary to convert an Object to an instance of this XSType.
277      */
278     public abstract String createFromJavaObjectCode(final String variableName);
279 
280     //--------------------------------------------------------------------------
281 
282     /**
283      * Creates source code for an additional constructor that deals with default
284      * values as specified in the XML schema instance.
285      * @param variableName Name of the constructor argument.
286      * @return Source code for dealing with default values.
287      */
288     public String createDefaultValueWithString(final String variableName) {
289         return " new " + getJType() + "(" + variableName + ")";
290     }
291     
292     /**
293      * Reads and sets the facets for XSType.
294      * 
295      * @param simpleType The SimpleType containing the facets.
296      */
297     public final void setFacets(final SimpleType simpleType) {
298         Enumeration<Facet> enumeration = simpleType.getEffectiveFacets();
299         while (enumeration.hasMoreElements()) {
300             setFacet(enumeration.nextElement());
301         }
302     }
303 
304     /**
305      * Set the given facet for XSType if applicable.
306      * 
307      * @param facet The facet to set for XSType.
308      */
309     protected abstract void setFacet(final Facet facet);
310     
311     /**
312      * Creates the validation code for an instance of this XSType. If necessary the validation
313      * code should create a newly configured TypeValidator, that should then be added to a
314      * FieldValidator instance whose name is provided.
315      *
316      * @param jsc The JSourceCode to fill in.
317      * @param fixedValue A fixed value to use if any.
318      * @param validatorInstanceName The name of the FieldValidator that the configured
319      *        TypeValidator should be added to.
320      */
321     public abstract void validationCode(final JSourceCode jsc,
322             final String fixedValue, final String validatorInstanceName);
323 
324     //--------------------------------------------------------------------------
325 }
