55  /**
56   * Describe an XML schema Century
57   * <p>
58   * The date type is derived from time period by setting up the facet:
59   * <ul>
60   * <li>duration to "P100Y"</li>
61   * </ul>
62   * <p>
63   * Note: This datatype is not included in any recommendation. It was introduced
64   * in http://www.w3.org/TR/2000/WD-xmlschema-2-20000407/ and was last in
65   * http://www.w3.org/TR/2000/CR-xmlschema-2-20001024/ and was removed by
66   * http://www.w3.org/TR/2001/PR-xmlschema-2-20010316/. It was not in the final
67   * approved recommendation: http://www.w3.org/TR/2001/REC-xmlschema-2-20010502/
68   *
69   * @author <a href="mailto:blandin@intalio.com">Arnaud Blandin</a>
70   * @version $Revision$
71   * @deprecated since Castor 1.0.6 since this type is not in any recommendation.
72   */
73  public class Century extends TimePeriod {
74  
75      /** SerialVersionUID */
76      private static final long serialVersionUID = 88787876938390034L;
77      /** The month format used by the toDate() method */
78      private static final String YEAR_FORMAT = "yyyy";
79  
80      /** Set to true and recompile to include debugging code in class. */
81      private static final boolean DEBUG = false;
82  
83      public Century() {
84          super("P100Y");
85          //we need to set the time zone to the computer local time zone
86          //if we want to use the toDate() method.
87          int temp = TimeZone.getDefault().getRawOffset();
88          if (temp < 0) {
89              temp = -temp;
90              try {
91                  this.setZoneNegative();
92              } catch (UnsupportedOperationException e) {
93                  //we are sure that we are dealing with a Century type
94                  //so we can never reach that point
95              }
96          }
97  
98          try {
99              short zhour = (short) (temp / (60*60*1000));
100             temp = temp % (60*60*1000);
101             short zmin = (short)(temp / (60*1000));
102             super.setZone(zhour, zmin);
103         } catch (UnsupportedOperationException e) {
104             //we are sure that we are dealing with a Century type
105             //so we can never reach that point
106         }
107     }
108 
109     /**
110      * convert this Year to a string
111      * The format is defined by W3C XML Schema draft and ISO8601
112      * i.e (+|-)CC
113      * @return a string representing this Century
114      */
115      public String toString() {
116         StringBuffer result = new StringBuffer();
117 
118         result.append(String.valueOf(this.getCentury()));
119         if (result.length() == 1) {
120             result.insert(0,0);
121         }
122 
123         if (isNegative()) {
124             result.insert(0,"-");
125         }
126         return result.toString();
127     }//toString

129    /**
130     * parse a String and convert it into a java.lang.Object
131     * @param str the string to parse
132     * @return the java.lang.Object represented by the string
133     * @throws ParseException a parse exception is thrown if the string to parse
134     *                        does not follow the rigth format (see the description
135     *                        of this class)
136     */
137     public static Object parse(String str) throws ParseException {
138         return parseCentury(str);
139     }
140 
141     /**
142      * parse a String and convert it into a Century
143      * @param str the string to parse
144      * @return the Century represented by the string
145      * @throws ParseException a parse exception is thrown if the string to parse
146      *                        does not follow the rigth format (see the description
147      *                        of this class)
148      */
149     public static Century parseCentury(String str) throws ParseException {
150         Century result = new Century();
151 
152         if (str.startsWith("-")) {
153             result.setNegative();
154             str =str.substring(1);
155         }
156 
157         if (DEBUG) {
158             System.out.println("In parsing method of Century");
159             System.out.println("String to parse : "+str);
160             System.out.println("Negative ? "+result.isNegative());
161         }
162 
163         if (str.length() != 2) {
164             throw new ParseException(str+": Bad XML Schema Century type format (CC)",0);
165         }
166 
167         if (DEBUG) {
168             System.out.println("Processing century: "+str.substring(0,2));
169         }
170         result.setCentury(Short.parseShort(str));
171 
172         return result;
173     } //parse
174 
175     public java.util.Date toDate() throws ParseException {
176         java.util.Date date = null;
177         SimpleDateFormat df = new SimpleDateFormat(YEAR_FORMAT);
178         SimpleTimeZone timeZone = new SimpleTimeZone(0,"UTC");
179 
180         // Set the time zone
181         if ( !isUTC() ) {
182             int offset = 0;
183             offset = ( (this.getZoneMinute() + this.getZoneHour()*60)*60*1000);
184             offset = isZoneNegative() ? -offset : offset;
185             timeZone.setRawOffset(offset);
186             timeZone.setID(TimeZone.getAvailableIDs(offset)[0]);
187         }
188         df.setTimeZone(timeZone);
189         date = df.parse(this.toString());
190         return date;
191     } //toDate()
