54  /**
55   * <p>
56   * Describe an XML schema Year.
57   * <p>
58   * The date type is derived from time period by setting up the facet :
59   * <ul>
60   * <li>duration to "P1Y"</li>
61   * </ul>
62   * <p>
63   * Note: This datatype is not included in any recommendation. It was introduced
64   * in http://www.w3.org/TR/2000/WD-xmlschema-2-20000407/ and was last in
65   * http://www.w3.org/TR/2000/CR-xmlschema-2-20001024/ and was removed by
66   * http://www.w3.org/TR/2001/PR-xmlschema-2-20010316/. It was not in the final
67   * approved recommendation: http://www.w3.org/TR/2001/REC-xmlschema-2-20010502/
68   *
69   * @author <a href="mailto:blandin@intalio.com">Arnaud Blandin</a>
70   * @version $Revision$
71   * @deprecated since Castor 1.0.6 since this type is not in any recommendation.
72   */
73  public class Year extends TimePeriod {
74      /** SerialVersionUID */
75      private static final long serialVersionUID = 8485456751196062574L;
76  
77      /** Set to true and recompile to include debugging code in class. */
78      private static final boolean DEBUG = false;
79  
80      /** The month format used by the toDate() method */
81      private static final String YEAR_FORMAT = "yyyy";
82  
83      public Year() {
84          super("P1Y");
85          //we need to set the time zone to the computer local time zone
86          //if we want to use the toDate() method.
87          int temp = TimeZone.getDefault().getRawOffset();
88          if (temp < 0){
89              temp = -temp;
90              try {
91                  this.setZoneNegative();
92              } catch(UnsupportedOperationException e) {
93              }
94          }
95          short zhour = (short) (temp / (60*60*1000));
96          temp = temp % (60*60*1000);
97          short zmin = (short)(temp / (60*1000));
98          try {
99              this.setZone(zhour,zmin);
100         } catch (UnsupportedOperationException e) {
101         }
102     }

155     /**
156      * convert this Year to a string
157      * The format is defined by W3C XML Schema draft and ISO8601
158      * i.e (+|-)CCYY
159      * @return a string representing this Month
160      */
161      public String toString() {
162 
163         StringBuffer result = new StringBuffer();
164         result.append(this.getCentury());
165         if (result.length() == 1)
166             result.insert(0,0);
167 
168         if ((this.getYear()/10) == 0)
169             result.append(0);
170         result.append(this.getYear());
171 
172         if (isNegative())
173             result.insert(0,'-');
174 
175         return result.toString();
176 
177     }//toString

179     /**
180     * parse a String and convert it into a java.lang.Object
181     * @param str the string to parse
182     * @return the java.lang.Object represented by the string
183     * @throws ParseException a parse exception is thrown if the string to parse
184     *                        does not follow the rigth format (see the description
185     *                        of this class)
186     */
187     public static Object parse(String str) throws ParseException {
188         return parseYear(str);
189     }
190 
191     /**
192      * parse a String and convert it into a Year
193      * @param str the string to parse
194      * @return the Year represented by the string
195      * @throws ParseException a parse exception is thrown if the string to parse
196      *                        does not follow the rigth format (see the description
197      *                        of this class)
198      */
199     public static Year parseYear(String str) throws ParseException {
200 
201         Year result = new Year();
202 
203         if ( str.startsWith("-") ) {
204             result.setNegative();
205             str =str.substring(1);
206         }
207 
208         if (DEBUG) {
209             System.out.println("In parsing method of Year");
210             System.out.println("String to parse : "+str);
211             System.out.println("Negative ? "+result.isNegative());
212         }
213 
214 
215         if (str.length() != 4)
216             throw new ParseException(str+": Bad XML Schema Year type format (CCYY)",0);
217 
218         if (DEBUG) {
219             System.out.println("Processing century: "+str.substring(0,2));
220         }
221         result.setCentury(Short.parseShort( str.substring(0,2) ));
222         if (DEBUG) {
223             System.out.println("Processing year: "+str.substring(2,4));
224         }
225         try {
226             result.setYear(Short.parseShort( str.substring(2,4) ));
227         } catch(UnsupportedOperationException e) {
228         }
229         return result;
230     }//parse

232      public java.util.Date toDate() throws ParseException {
233         java.util.Date date = null;
234         SimpleDateFormat df = new SimpleDateFormat(YEAR_FORMAT);
235         SimpleTimeZone timeZone = new SimpleTimeZone(0,"UTC");
236 
237         // Set the time zone
238         if ( !isUTC() ) {
239             int offset = 0;
240             offset = ( (this.getZoneMinute() + this.getZoneHour()*60)*60*1000);
241             offset = isZoneNegative() ? -offset : offset;
242             timeZone.setRawOffset(offset);
243             timeZone.setID(TimeZone.getAvailableIDs(offset)[0]);
244         }
245         df.setTimeZone(timeZone);
246         date = df.parse(this.toString());
247         return date;
248     }//toDate()
