50  package org.exolab.castor.util;
51  
52  import java.io.IOException;
53  import java.io.ObjectInputStream;
54  import java.io.ObjectOutputStream;
55  import java.io.Serializable;
56  import java.util.EventListener;
57  
58  /**
59   * <p>This class is an efficient repository for EventListeners based
60   * on javax.swing.EventListenerList.</p>
61   *
62   * <p>This modification of javax.swing.EventListenerList retains the
63   * core functionality of that class but changes the basic API and adds
64   * a few more features, as summarized below:<p>
65   * 
66   * <ol>
67   * <li>javax.swing.EventListenerList requires all listeners to be
68   * added in conjunction with a class object that identified the type
69   * of the listener.  This implementation's add methods simply take the
70   * listener object.
71   *
72   * <li>The listener list returned from javax.swing.EventListenerList
73   * had to be iterated over in a cumbersome manner because the
74   * listeners were stored along with their Class objects in the array.
75   * Since listener classes are not stored in this listener list, the
76   * returned listener array can be iterated over normally (1 element at
77   * a time).
78   *
79   * <li>The remove method in javax.swing.EventListenerList had return
80   * type "void".  This implementation's remove method returns true if
81   * the specified listener was found in the listener array and false
82   * otherwise.
83   *
84   * <li>This implementation adds {@link #add(EventListener, int)},
85   * which allows the addition of a listener at a specific position in
86   * the array.  
87   *
88   * <li>The add and remove methods in this implementation throw
89   * IllegalArgumentExceptions when their arguments are null.
90   * </ol>
91   *
92   * <p>As is the case with javax.swing.EventListenerList, this class
93   * provides multi-threaded safety through a copy-on-modify strategy.
94   * It is optimized to provide high performance when events are being
95   * fired, with slightly slower performance than the Collection API
96   * when listeners are being added and removed.  Like its predecessor,
97   * this class will never return a null array from getListenerList.</p>
98   *
99   * <p>The most important thing to keep in mind when using this class
100  * is that the array returned by getListenerList is the actual
101  * internal array of this class and MUST NOT BE MODIFIED UNDER ANY
102  * CIRCUMSTANCES.  Below is an example of how to use this class,
103  * borrowed (and slightly modified) from the
104  * javax.swing.EventListenerList documentation:</p>
105  *
106  * <p>Usage example:
107  *    Say one is defining a class that sends out FooEvents, and one wants
108  * to allow users of the class to register FooListeners and receive 
109  * notification when FooEvents occur.  The following should be added
110  * to the class definition:</p>
111  * <pre>
112  * EventListenerList listenerList = new EventListenerList();
113  * FooEvent fooEvent = null;
114  *
115  * public void addFooListener(FooListener l) {
116  *     listenerList.add(l);
117  * }
118  *
119  * public void removeFooListener(FooListener l) {
120  *     listenerList.remove(l);
121  * }
122  *
123  * // Notify all listeners that have registered interest for
124  * // notification on this event type.  The event instance 
125  * // is lazily created using the parameters passed into 
126  * // the fire method.
127  *
128  * protected void fireFooXXX() {
129  *     // Guaranteed to return a non-null array
130  *     EventListener[] listeners = listenerList.getListenerList();
131  *     // Process the listeners last to first, notifying
132  *     // those that are interested in this event
133  *     for (int i = 0 ; i &lt; listeners.length ; i++) {
134  *         // Lazily create the event:
135  *         if (fooEvent == null)
136  *             fooEvent = new FooEvent(this);
137  *         ((FooListener)listeners[i]).fooXXX(fooEvent);
138  *     }
139  * }
140  * </pre>
141  *
142  * <p>foo should be changed to the appropriate name, and fireFooXxx to the
143  * appropriate method name.  One fire method should exist for each
144  * notification method in the FooListener interface.</p>
145  *
146  * <p>The authors of javax.swing.EventListenerList are Georges Saab,
147  * Hans Muller, and James Gosling.</p>
148  *
149  * @author <a href="mailto:Jeff.Norris@jpl.nasa.gov">Jeff Norris</a>
150  * @version $Revision$ $Date: 2005-12-13 14:58:48 -0700 (Tue, 13 Dec 2005) $
151  */
152 public class EventListenerList implements Serializable {
153     /** SerialVersionUID */
154     private static final long serialVersionUID = 4472874989562384564L;
155 
156   /**
157    * A null array to be shared by all empty listener lists
158    */
159   private final static EventListener[] NULL_ARRAY = new EventListener[0];
160   
161   /** 
162    * The internal list of listeners that is returned from
163    * getListenerList 
164    */
165   protected transient EventListener[] listenerList = NULL_ARRAY;
166 
167   /**
168    * <p>Passes back the event listener list as an array of
169    * EventListeners.</p>
170    *
171    * <p>Note that for performance reasons, this implementation passes
172    * back the actual data structure in which the listener data is
173    * stored internally!  This method is guaranteed to pass back a
174    * non-null array, so that no null-checking is required in fire
175    * methods.  A zero-length array of Object will be returned if
176    * there are currently no listeners.</p>
177    * 
178    * WARNING!!! Absolutely NO modification of
179    * the data contained in this array should be made -- if
180    * any such manipulation is necessary, it should be done
181    * on a copy of the array returned rather than the array 
182    * itself.
183    */
184   public EventListener[] getListenerList() {
185     return listenerList;
186   }
187 
188   /**
189    * <p>Returns the total number of listeners in this listener
190    * list.</p>
191    *
192    */
193   public int getListenerCount() {
194     return listenerList.length;
195   }
196 
197   /**
198    * <p>Adds the listener to the end of the listener list.</p>
199    *
200    * @param newListener the listener to be added
201    * @exception IllegalArgumentException if the specified newListener
202    * is null
203    */
204   public synchronized void add(EventListener newListener) 
205   {
206     
207     if (newListener==null) 
208       throw new IllegalArgumentException("Listener to add must not be null.");
209     
210     if (listenerList == NULL_ARRAY) {
211 	    // if this is the first listener added, 
212 	    // initialize the lists
213 	    listenerList = new EventListener[] { newListener };
214     } else {
215 	    // Otherwise copy the array and add the new listener
216 	    int oldLength = listenerList.length;
217 	    EventListener[] tmp = new EventListener[oldLength+1];
218 	    System.arraycopy(listenerList, 0, tmp, 0, oldLength);
219 
220 	    tmp[oldLength] = newListener;
221 
222 	    listenerList = tmp;
223     }
224   }
225 
226   /**
227    * <p>Adds the listener at the specified index in the listener
228    * list.</p>
229    *
230    * @param newListener the listener to be added
231    * @exception IllegalArgumentException if the specified newListener
232    * is null, or the specified index is less than zero or greater than
233    * the length of the listener list array.
234    */
235   public synchronized void add(EventListener newListener, int index) 
236   {
237     if (newListener==null) 
238       throw new IllegalArgumentException("Listener to add must not be null.");
239     
240     if ((index < 0) || (index > listenerList.length))
241       throw new IllegalArgumentException("Index to add listener (" + index + 
242                                          ") is out of bounds. List length is " +
243                                          listenerList.length);
244     
245     if (listenerList == NULL_ARRAY) {
246 	    // if this is the first listener added, initialize the lists
247 	    listenerList = new EventListener[] { newListener };
248     } else {
249       // Otherwise copy the array and add the new listener
250       int oldLength = listenerList.length;
251 	    EventListener[] tmp = new EventListener[oldLength+1];
252       // Copy up to the index where the new listener should go
253 	    System.arraycopy(listenerList, 0, tmp, 0, index);
254       // Skip a cell and copy the rest of the list
255       System.arraycopy(listenerList, index, tmp, index+1, oldLength-index);
256       // Insert the new listener
257 	    tmp[index] = newListener;
258 
259 	    listenerList = tmp;
260     }
261   }
262 
263   /**
264    * Removes the listener as a listener of the specified type.
265    *
266    * @param listenerToRemove the listener to be removed
267    * @exception IllegalArgumentException if the specified listener is
268    * null
269    */
270   public synchronized boolean remove(EventListener listenerToRemove) {
271     if (listenerToRemove ==null)
272       throw new IllegalArgumentException("Listener to remove must " +
273                                          "not be null.");
274     // Is listenerToRemove on the list?
275     int index = -1;
276     for (int i = listenerList.length-1; i>=0; i--) {
277 	    if (listenerList[i].equals(listenerToRemove) == true) {
278         index = i;
279         break;
280 	    }
281     }
282 
283     // If so,  remove it
284     if (index != -1) {
285 	    EventListener[] tmp = new EventListener[listenerList.length-1];
286 	    // Copy the list up to index
287 	    System.arraycopy(listenerList, 0, tmp, 0, index);
288 	    // Copy from one past the index, up to the end of tmp (which is
289 	    // one element shorter than the old list)
290 	    if (index < tmp.length)
291         System.arraycopy(listenerList, index+1, tmp, index, 
292                          tmp.length - index);
293 	    // set the listener array to the new array or null
294 	    listenerList = (tmp.length == 0) ? NULL_ARRAY : tmp;
295       return true;
296     }
297     return false;
298   }
299 
300   // Serialization support.  
301   private void writeObject(ObjectOutputStream s) throws IOException {
302     Object[] lList = listenerList;
303     s.defaultWriteObject();
304 	
305     // Save the non-null event listeners:
306     for (int i = 0; i < lList.length; i+=1) {
307 	    EventListener l = (EventListener)lList[i];
308 	    if ((l!=null) && (l instanceof Serializable)) {
309         s.writeObject(l);
310 	    }
311     }
312 	
313     s.writeObject(null);
314   }
315 
316   private void readObject(ObjectInputStream s) 
317     throws IOException, ClassNotFoundException {
318     listenerList = NULL_ARRAY;
319     s.defaultReadObject();
320     EventListener listenerOrNull;
321     
322     while (null != (listenerOrNull = (EventListener)s.readObject())) {
323       add(listenerOrNull);
324     }	    
325   }
326   
327   /**
328    * Returns a string representation of the EventListenerList.
329    */
330   public String toString() {
331     Object[] lList = listenerList;
332     String s = "EventListenerList: ";
333     s += lList.length + " listeners: ";
334     for (int i = 0 ; i < lList.length ; i++) {
335 	    s += " listener " + lList[i+1];
336     }
337     return s;
338   }
