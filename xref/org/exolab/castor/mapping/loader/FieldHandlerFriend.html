42  /**
43   * An extended version of the FieldHandler interface which is used for adding additional
44   * functionality while preserving backward compatability.
45   * 
46   * @author <a href="kvisco@intalio.com">Keith Visco</a>
47   * @version $Revision$ $Date: 2005-08-03 15:11:51 -0600 (Wed, 03 Aug 2005) $
48   * @see FieldDescriptor
49   * @see FieldHandler
50   */
51  public abstract class FieldHandlerFriend<T> implements FieldHandler<T> {
52  
53    /**
54     * Returns the FieldDescriptor for the field that this handler is responsible for, or null if no
55     * FieldDescriptor has been set. This method is useful for implementations of the
56     * {@link FieldHandler} interface that wish to obtain information about the field in order to make
57     * the FieldHandler more generic and reusable, or simply for validation purposes.
58     * 
59     * @return the {@link FieldDescriptor}, or null if none exists.
60     */
61    protected abstract FieldDescriptor getFieldDescriptor();
62  
63    /**
64     * Sets the FieldDescriptor that this FieldHander is responsibile for. By setting the
65     * FieldDescriptor, it allows the implementation of the FieldHandler methods to obtain information
66     * about the field itself. This allows a particular implementation to become more generic and
67     * reusable.
68     * 
69     * @param fieldDesc the FieldDescriptor to set
70     */
71    public abstract void setFieldDescriptor(FieldDescriptor fieldDesc);
