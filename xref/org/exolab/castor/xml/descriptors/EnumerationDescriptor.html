35  package org.exolab.castor.xml.descriptors;
36  
37  import org.exolab.castor.mapping.FieldHandler;
38  import org.exolab.castor.xml.NodeType;
39  import org.exolab.castor.xml.XMLFieldHandler;
40  import org.exolab.castor.xml.util.XMLClassDescriptorImpl;
41  import org.exolab.castor.xml.util.XMLFieldDescriptorImpl;
42  
43  /**
44   * The default class descriptor for Enumerations that are passed in as the root-level class, this is
45   * really only useful for marshaling, as Enumerations are immutable.
46   *
47   * @author <a href="mailto:kvisco-at-intalio.com">Keith Visco</a>
48   * @version $Revision$ $Date: 2004-12-16 22:49:25 -0700 (Thu, 16 Dec 2004) $
49   */
50  public class EnumerationDescriptor extends XMLClassDescriptorImpl {
51  
52    // ----------------/
53    // - Constructors -/
54    // ----------------/
55  
56    /**
57     * Creates a new EnumerationDescriptor.
58     */
59    public EnumerationDescriptor() {
60      super(java.util.Enumeration.class);
61      // -- create element descriptor
62  
63      XMLFieldDescriptorImpl desc =
64          new XMLFieldDescriptorImpl(Object.class, "_elements", null, NodeType.Element);
65  
66      FieldHandler handler = (new XMLFieldHandler() {
67  
68        /**
69         * {@inheritDoc}
70         */
71        public Object getValue(final Object object) throws IllegalStateException {
72          return object;
73        }
74  
75        /**
76         * {@inheritDoc}
77         */
78        public void setValue(final Object object, final Object value)
79            throws IllegalStateException, IllegalArgumentException {
80          // -- do nothing since enumerations are immutable.
81        }
82  
83        /**
84         * {@inheritDoc}
85         */
86        public Object newInstance(final Object parent) {
87          // -- not used
88          return null;
89        }
90      });
91      desc.setHandler(handler);
92      desc.setMultivalued(true);
93  
94      addFieldDescriptor(desc);
95    } // -- EnumerationDescriptor()
96  
97  } // -- EnumerationDescriptor
