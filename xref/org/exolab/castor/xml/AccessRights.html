36  package org.exolab.castor.xml;
37  
38  /**
39   * A class used to indicate access rights
40   * 
41   * @author <a href="kvisco@intalio.com">Keith Visco</a>
42   * @version $Revision$ $Date: 2003-03-03 00:05:44 -0700 (Mon, 03 Mar 2003) $
43   **/
44  public class AccessRights {
45  
46    /**
47     * The type that indicates both read and write access
48     **/
49    public static final short BOTH = 0;
50  
51    /**
52     * The type that indicates only read access
53     **/
54    public static final short READ = 1;
55  
56    /**
57     * The type that indicates only write access
58     **/
59    public static final short WRITE = 2;
60  
61  
62    /**
63     * A read and write AccessRights
64     **/
65    public static final AccessRights both = new AccessRights(AccessRights.BOTH);
66  
67    /**
68     * A read-only AccessRights
69     **/
70    public static final AccessRights read = new AccessRights(AccessRights.READ);
71  
72    /**
73     * A write-only AccessRights
74     **/
75    public static final AccessRights write = new AccessRights(AccessRights.WRITE);
76  
77    /**
78     * The type of this AccessRights
79     **/
80    private short type = BOTH;
81  
82    /**
83     * Creates a default AccessRights with both read and write privileges
84     **/
85    private AccessRights() {
86      super();
87    } // -- AccessRights
88  
89    /**
90     * Creates a new AccessRights with the privilege set to the given type
91     * 
92     * @param type the type of privilege to be granted
93     **/
94    private AccessRights(short type) {
95      this.type = type;
96    } // -- AccessRights
97  
98    /**
99     * Returns the type of this Access
100    **/
101   public short getType() {
102     return type;
103   } // -- getType
104 
105 
106   /**
107    * Returns true if this AccessRights allows reading
108    * 
109    * @return true if this AccessRights allows reading, otherwise false.
110    **/
111   public boolean isReadable() {
112     return ((type == BOTH) || (type == READ));
113   } // -- isReadable
114 
115   /**
116    * Returns true if this AccessRights allows writing
117    * 
118    * @return true if this AccessRights allows writing, otherwise false.
119    **/
120   public boolean isWritable() {
121     return ((type == BOTH) || (type == WRITE));
122   } // -- isWritable
123 
124 
125 } // -- AccessRights
