48  
49  /**
50   * Package private class to handle the case where a simple type
51   * can't be instanciated because its parent has not yet been read.
52   *
53   * @author <a href="mailto:berry@intalio.com">Arnaud Berry</a>
54   * @version $Revision:
55  **/
56  class DeferredSimpleType extends SimpleType {
57      /** SerialVersionUID */
58      private static final long serialVersionUID = 1335439479275580848L;
59  
60      /**
61       * An instance of ListType or of a class derived from AtomicType
62       * created the first time getType is called from the information contained
63       * in this simple type instance.
64      **/
65      private SimpleType instance= null;
66  
67      /**
68       * The parent's name.
69      **/
70      private String baseTypeName = null;
71  
72      /**
73       * The number of times createInstance was called (to guess when there is a cyclic
74       * reference (type A extends B; type B extends A) in the schema types)
75       * TODO: real cycle detection.
76       */
77      private int createInstanceCallsCount= 0;
78  
79      /**
80       * Creates a new default DeferredSimpleType.
81      **/
82      DeferredSimpleType() {
83          super();
84      } //-- DeferredSimpleType
85      
86      /**
87       * Creates a new DeferredSimpleType
88       *
89       * @param name the name of the SimpleType.
90      **/
91      DeferredSimpleType(String name) {
92          super();
93          setName(name);
94      } //-- DeferredSimpleType
95      
96      /**
97       * Creates a new DeferredSimpleType
98       *
99       * @param name the name of the SimpleType.
100      * @param schema the parent Schema of the SimpleType.
101     **/
102     DeferredSimpleType(String name, Schema schema) {
103         setName(name);
104         setSchema(schema); 
105     } //-- DeferredSimpleType
106     
107     /**
108      * Sets the name of the base type that couldn't be resolved
109      * when this type was created.
110      */
111     public void setBaseTypeName(String baseTypeName) {
112         this.baseTypeName= baseTypeName;
113     }
114 
115     /**
116      * Package private getter of the simpleType instance.
117      */
118     XMLType getType() {
119         if (instance == null) createInstance();
120         return instance;
121     }
122 
123 
124     /**
125      * Resolves the parents of this simple type in order to create the
126      * instance of the appropriate class deriving from simple type and fill it.
127      */
128     protected synchronized void createInstance()
129     {
130         createInstanceCallsCount++;
131         if (createInstanceCallsCount >= 666) {
132             String err = "cyclic type definition involving the type: " + getName();
133             throw new IllegalStateException(err);
134         }
135 
136         //create the type, false means we don't want a DeferredSimpleType to be returned.
137         instance= Schema.getTypeFactory().createUserSimpleType( getSchema(),
138                                                                 getName(),
139                                                                 baseTypeName,
140                                                                 getDerivationMethod(),
141                                                                 false);
142 
143         if (instance != null) {
144            copyFacets(instance);
145            instance.setParent(getParent());
146         }
147     }
148 
149     /**
150      * Returns Structure.UNKNOWN
151      * (This class should not be seen outside AttributeDecl and ElementDecl anyway)
152     **/
153     public short getStructureType() {
154         return Structure.UNKNOWN;
155     } //-- getStructureType
156     
157     /**
158      * Overridden to return the local copy of the base type's name we own.
159      * @see XMLType#getBaseTypeName()
160      */
161     public String getBaseTypeName() {
162         return baseTypeName;
163     }
164 }
