33  package org.exolab.javasource;
34  
35  import org.apache.commons.lang3.StringUtils;
36  
37  /**
38   * A class which holds information about a field. Modeled closely after the Java Reflection API.
39   * This class is part of package which is used to create source code in memory.
40   *
41   * @author <a href="mailto:keith AT kvisco DOT com">Keith Visco</a>
42   * @version $Revision$ $Date: 2005-02-26 17:30:28 -0700 (Sat, 26 Feb 2005) $
43   */
44  public final class JField extends AbstractJField {
45  
46    /**
47     * Creates a new JField.
48     * 
49     * @param type JType of this new field.
50     * @param name Name of this new field.
51     */
52    public JField(final JType type, final String name) {
53      this(type, name, "Field " + name + ".");
54    }
55  
56    public JField(final JType type, final String name, String comment) {
57      super(type, name);
58  
59      JModifiers modifiers = getModifiers();
60      modifiers.makePrivate();
61  
62      if (StringUtils.isNotEmpty(comment)) {
63        JDocComment jDocComment = new JDocComment();
64        jDocComment.appendComment(comment);
65        setComment(jDocComment);
66      }
67    }
68  
69  }
