35  /**
36   * An interface which represents a Member of a JClass, modelled closely after the Java Reflection
37   * API. This class is part of a package used to represent source code.
38   *
39   * @author <a href="mailto:keith AT kvisco DOT com">Keith Visco</a>
40   * @version $Revision$ $Date: 2005-02-26 17:30:28 -0700 (Sat, 26 Feb 2005) $
41   */
42  public interface JMember {
43  
44    /**
45     * Returns the modifiers for this JMember.
46     * 
47     * @return The modifiers for this JMember.
48     */
49    JModifiers getModifiers();
50  
51    /**
52     * Returns the name of this JMember.
53     * 
54     * @return The name of this JMember.
55     */
56    String getName();
57  
58  }
