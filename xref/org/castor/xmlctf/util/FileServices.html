45  package org.castor.xmlctf.util;
46  
47  import java.io.File;
48  import java.io.FileInputStream;
49  import java.io.FileOutputStream;
50  import java.io.IOException;
51  import java.io.InputStream;
52  import java.io.OutputStream;
53  import java.util.Enumeration;
54  import java.util.jar.JarFile;
55  import java.util.zip.ZipEntry;
56  
57  /**
58   * This class is a set of tools for manipulating files needed by the CTF.
59   *
60   * @author <a href="mailto:blandin@intalio.com">Arnaud Blandin</a>
61   * @version $Revision: 6787 $ $Date: 2005-03-05 06:42:06 -0700 (Sat, 05 Mar 2005) $
62   */
63  public final class FileServices {
64  
65      /** The string for a CVS subdirectory. */
66      public static final String CVS  = "CVS";
67      /** The string for a Subversion subdirectory. */
68      public static final String SVN  = ".svn";
69      /** The string for a git subdirectory. */
70      public static final String GIT  = ".git";
71      /** The string for an extension for a XSD file. */
72      public static final String XSD  = ".xsd";
73      /** The string for an extension for a XML file. */
74      public static final String XML  = ".xml";
75      /** The string for an extension for a Java file. */
76      public static final String JAVA = ".java";
77      /** The string for an extension for a JAR file. */
78      public static final String JAR  = ".jar";
79  
80      /** Buffer size used for copying files. */
81      private static final int BUF_SIZE = 16 * 1024;
82  
83      /**
84       * No-arg constructor.  Private as we're a utility class.
85       */
86      private FileServices() {
87          // Nothing to do
88      }
89  
90      /**
91       * Copy all the needed documents (java, xsd, xml file) of given file (jar or
92       * directory) to a specified directory.
93       *
94       * @param file the file that contains the entries to copy
95       * @param root the destination directory to copy files to
96       * @throws IOException if an error occurs while copying files
97       */
98      public static void copySupportFiles(final File file, final File root) throws IOException {
99          if (file == null) {
100             throw new IllegalArgumentException("The Entry file is null");
101         }
102         if (root == null) {
103             throw new IllegalArgumentException("The destination directory is null");
104         }
105 
106         if (file.isDirectory()) {
107             copySupportFilesForDirectory(file, root);
108         } else if (file.getName().endsWith(JAR)) {
109             copySupportFilesForJarFile(file, root);
110         } else {
111             //ignore other file type
112         }
113     }
114 
115     /**
116      * Unzips support files out of the JAR to our output directory. See
117      * {@link #isSupportFile(String)} for the definition of support file. This
118      * method does not totally unzip the JAR file. It only extracts certain
119      * important files out of the JAR.
120      *
121      * @param file the file that contains the entries to copy
122      * @param root the destination directory to copy files to
123      * @throws IOException if an error occurs while copying files
124      */
125     private static void copySupportFilesForJarFile(final File file, final File root)
126                                                                       throws IOException {
127         JarFile jar = new JarFile(file);
128         for (Enumeration e = jar.entries(); e.hasMoreElements(); ) {
129             ZipEntry entry = (ZipEntry) e.nextElement();
130             if (isSupportFile(entry.getName())) {
131                 InputStream src = jar.getInputStream(entry);
132                 File out = new File(root, entry.getName());
133                 out.getParentFile().mkdirs();
134                 copy(src, new FileOutputStream(out));
135                 src.close();
136             }
137         }
138     }
139 
140     /**
141      * Copies support files from one directory to our output directory,
142      * recursing into other directories. See {@link #isSupportFile(String)} for
143      * the definition of support file.
144      *
145      * @param file the file that contains the entries to copy
146      * @param root the destination directory to copy files to
147      * @throws IOException if an error occurs while copying files
148      */
149     private static void copySupportFilesForDirectory(final File file, final File root)
150                                                                       throws IOException {
151         File[] entries = file.listFiles();
152         for (int i = 0; i < entries.length; i++) {
153             File tempEntry = entries[i];
154             if (isSupportFile(tempEntry.getName())) {
155                 File out = new File(root, tempEntry.getName());
156                 out.getParentFile().mkdir();
157                 FileOutputStream dest = new FileOutputStream(out);
158 
159                 InputStream src = new FileInputStream(tempEntry);
160                 copy(src, dest);
161                 src.close();
162                 dest.close();
163             } else if (tempEntry.isDirectory()) {
164                 File out = new File(root, tempEntry.getName());
165                 out.mkdir();
166                 copySupportFiles(tempEntry, out);
167             }
168         }
169     }
170 
171     /**
172      * Copies an InputStream into a OutputStream.
173      *
174      * @param src Source input stream
175      * @param dest Destination output stream
176      * @throws IOException if an error occurs while copying files
177      */
178     private static void copy(final InputStream src, final OutputStream dest) throws IOException {
179         byte[] buf = new byte[BUF_SIZE];
180         int read;
181         while (true) {
182             read = src.read(buf, 0, BUF_SIZE);
183             if (read == -1) {
184                 break;
185             }
186             dest.write(buf, 0, read);
187         }
188     }
189 
190     /**
191      * Return true if the file is a support file for the test. A support file is
192      * a schema or a java file.
193      *
194      * @param name File name to check to see if it represents a support file
195      * @return true if the file is a support file for the test.
196      */
197     private static boolean isSupportFile(final String name) {
198         return name.endsWith(XSD) || name.endsWith(JAVA) || name.endsWith(XML);
199     }
200 
201     /**
202      * Return true if the file provided is a special source control management
203      * directory that we want to ignore.
204      * @param name Name of a directory
205      * @return true if the file provided is an SCM directory
206      */
207     public static boolean isScmDirectory(final String name) {
208         return name.endsWith(FileServices.CVS) || name.equals(FileServices.SVN);
209     }
210 
211 }
